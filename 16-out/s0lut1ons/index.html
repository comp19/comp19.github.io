import java.util.Scanner;

public class ExercicioUm {
	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);

		int n;
		do {
			System.out.println("Quandos números deseja inserir?");
			n = sc.nextInt();
		} while (n <= 0);  // Enquanto o nº inserido for negativo ou zero

		int i = 0;  // contador de números inseridos
		int num;  // variável que irá guardar os números inseridos.
		while (i < n) {

			System.out.println("Insira o " + (i+1) + "º número:");
			num = sc.nextInt();

			if (num % 2 == 0) {
				System.out.println("O número é par.");
			}
			else {
				System.out.println("O número é ímpar.");
			}

			i++;
		}

		sc.close();
	}
}

import java.util.Scanner;

public class ExercicioDois {

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        int n;
        do {
            System.out.println("Quantos números quer inserir?");
            n = sc.nextInt();
        } while (n<=0);

        double soma = 0;
        int i = 0;
        while (i < n) {
            System.out.println("Insira o "+(i+1)+"º número:");
            double num = sc.nextDouble();
            soma += num; //igual a : soma = soma + num;
            i++; //igual a: i = i + 1; ou ainda: i += 1;
        }
        sc.close();

        double media = soma / n;
        System.out.println("A média dos números inseridos é "+media);
    }
}
