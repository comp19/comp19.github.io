import java.util.Scanner;

public class RollTheDice {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);

        int retry_limit = 6;
        int attempts = 0;

        // resultado dos dados:
        int min = 1;  // valor mínimo dos dados
        int max = 12; // valor máximo dos dados
        int diceResult = (int) (Math.random() * ((max - min) + 1)) + min;

        while (attempts < retry_limit) {
            System.out.println("Please place your bet. You have "
                    + (retry_limit - attempts) + " attempts left:");
            int bet = scanner.nextInt();
            if (bet == diceResult) {
                // feedback para o utilizador - VENCEU
                System.out.println("Winner!");
                break;
            } else {
                // feedback para o utilizador - PERDEU
                System.out.println("Loser!");
            }
            attempts++;
        }
        scanner.close();
        System.out.println(diceResult);
        if (attempts >= 6) {
            System.out.println("You wasted all your attempts! Dumb ass!");
        }
    }
}
